 48#pragma once
 49
 50#include <memory>
 51#include <functional>
 52
 53// ROS imports
 54#include "rclcpp/rclcpp.hpp"
 55
 56// Pegasus imports 
 57#include "state.hpp"
 58
 59namespace autopilot {
 60
 61class Geofencing {
 62
 63public:
 64
 65    using SharedPtr = std::shared_ptr<Geofencing>;
 66    using UniquePtr = std::unique_ptr<Geofencing>;
 67    using WeakPtr = std::weak_ptr<Geofencing>;
 68
 69    // Configuration for the geofencing
 70    struct Config {
 71        rclcpp::Node::SharedPtr node;                                           // ROS 2 node ptr (in case the mode needs to create publishers, subscribers, etc.)
 72        std::function<State()> get_vehicle_state;                               // Function pointer to get the current state of the vehicle
 73        std::function<VehicleStatus()> get_vehicle_status;                      // Function pointer to get the current status of the vehicle
 74        std::function<VehicleConstants()> get_vehicle_constants;                // Function pointer to get the current constants of the vehicle
 75    };
 76
 77    // Custom constructor like function - as we must have the default constructor for the pluginlib
 78    inline void initialize_geofencing(const Geofencing::Config & config) {
 79        
 80        // Initialialize the node
 81        node_ = config.node;
 82
 83        // Initialize the function pointers
 84        get_vehicle_state_ = config.get_vehicle_state;
 85        get_vehicle_status_ = config.get_vehicle_status;
 86        get_vehicle_constants_ = config.get_vehicle_constants;
 87
 88        // Perform custom initialization
 89        initialize();
 90    }
 91
 92    /** @brief Initialize the custom geofencing object */
 93    virtual void initialize() = 0;
 94
 95    /** 
 96     * @brief Checks if a geofencing violation has ocurred.
 97     * @return true if a geofencing violation has ocurred, false otherwise
 98     */
 99    virtual bool check_geofencing_violation() = 0;
100
101protected:
102
103    // Node
104    rclcpp::Node::SharedPtr node_;
105
106    // Function pointers
107    std::function<State()> get_vehicle_state_;
108    std::function<VehicleStatus()> get_vehicle_status_;
109    std::function<VehicleConstants()> get_vehicle_constants_;
110};
111
112} // namespace autopilot
